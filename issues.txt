
```yaml
apiVersion: traefik.io/v1alpha1

metadata:
  name: auth-route
  namespace: auth-service
spec:
  entryPoints:
    - web
  routes:
    - match: Host(`auth.local`)
      kind: Rule
      services:
        - name: auth-service
          port: 8000
```

# ✅ STEP 2 — Create Ingress for auth-service

THIS is what you're missing.

Create:

```yaml
# services/auth-service/ingress.yaml
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: auth-ingress
  namespace: auth-service
  annotations:
    kubernetes.io/ingress.class: traefik
spec:
  rules:
  - host: auth.local
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: auth-service
            port:
              number: 8000
```
